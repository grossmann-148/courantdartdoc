    const Filters = (allOpus, artistFilter, availabilityFilter, typeFilter, sizeFilter, searchFilter) => {

  // Filter by artist
  const artistFilterFunction = (opus) => {
    if(!artistFilter.includes(opus.Artistname)) {
      opus.display = false;
    }
    return opus;
  }

  // Filter by availability
  const availabilityFilterFunction = (opus) => {
    if(!availabilityFilter.includes(opus.availability)) {
      opus.display = false;
    }
    return opus;
  }

  // Filter by type
  const typeFilterFunction = (opus) => {
    if(!typeFilter.includes(opus.TypeID)) {
      opus.display = false;
    }
    return opus;
  }

  // Filter by size
  const sizeFilterFunction = (opus) => {
    if (opus.Opuswidth < sizeFilter[0]) {
      opus.display = false;
    }
    if (opus.Opuswidth > sizeFilter[1]) {
      opus.display = false;
    }
    if (opus.Opusheigh < sizeFilter[2]) {
      opus.display = false;
    }
    if (opus.Opusheigh > sizeFilter[3]) {
      opus.display = false;
    }
    return opus;
  }

  // Filter by search
  const searchFilterFunction = (opus) => {
    const name = opus.Opusname.toLowerCase();
    if (name.search(searchFilter) === -1) {
      opus.display = false;
    }
    return opus;
  }
  allOpus.forEach(element => {
    element.display = true;
    if (artistFilter.length > 0) {
      element = artistFilterFunction(element);
    }
    if (availabilityFilter.length > 0) {
      element = availabilityFilterFunction(element);
    }
    if (typeFilter.length > 0) {
      element = typeFilterFunction(element);
    }
    if (searchFilter !== '') {
      element = searchFilterFunction(element);
    }
    element = sizeFilterFunction(element);
  });
  return allOpus;
}

export default Filters;
